// Everything else is managed with commands. //
    NOTE: You can pass 'all','others','me' in the username arg.

Commands:
    - r6 [username]             - R6's selected character.    
    - re [username]             - Respawns selected player.
    - jumpscare [username]      - Jumpscares selecter player.
    - kfc                       - Loads kfc map.
    - ciapanel [username]       - Loads the CIA panel.
    - 911                       - Loads a 9/11 script.
    - knife [username]          - Loads grab knife v4.
    - sugma [username]          - Loads project sugma.
    - ssp [username]            - Loads Secret Service Panel.
    - dex [username]		- Loads Dex.
    - kick [username]           - Kicks the selected player.
    - shutdown [reason]         - Shuts down the server. (ABUSIVE)
    - disco [username]          - Enables / disables disco mode.
    - shedletsky		- Shedletskifies everyone.
    - rc7 [username]            - Loads RC7 for a player.
    - invisible [username]      - Turns the selected player invisible.
    - visible [username]	- Turns selected player visible.
    - explode [username]        - Explodes selected player.
    - hdranker [username]       - Loads the HD Admin ranker.
    - char [player] [username]	- Chars selected player.
    - h [text]			- Shows a server hint.
    - m [text]			- Shows a server message.
    - permhint			- Same as "h", but the hint doesn't disappear.
    - gravityhammer [username]	- Loads the Gravity Hammer for selected player.
    - flamethrower [username]	- Loads the flamethrower.
    - magespells [username]	- Loads the mage spells script.
    - classictools [username]	- Loads the classic tools script.
    - guns [username]		- Loads a guns script.
    - masterchief [username]	- Loads a Master Chief script.
    - playid [soundid]		- Plays a sound for the server. 
    - stopsound			- Stops the current playing sound
    - clearlogs			- Clears the output logs.
    - kidnap [username]		- Kidnaps a player.
    - clone [username]		- Clones a player.
    - forcefield [username]	- Gives selected player a forcefield.
    - unforcefield [username]	- Removes forcefield from selected player.
    - c4 [username]		- Gives selected player a C4 remote.
    - LOHOO [any(required)]	- Nukes the server.
    - speedcoil [username]	- Gives selected player a speed coil.
    - gravitycoil [username]	- Gives selected player a gravity coil.
    - rainingtacos [any(req)]	- Enables raining tacos.
    - stoptacos [any(req)]      - Stops raining tacos.
    - sit [username]		- Makes the player sit.
